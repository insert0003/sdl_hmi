project(sdl_bt)

cmake_minimum_required(VERSION 2.8)

set(PROJECT_NAME "sdl_usb")
set(PROJECT_VERSION "1.0")
set(PROJECT_PRETTY_NAME "SDLUSB")
set(PROJECT_DESCRIPTION "sdl_usb hmi")

add_custom_target(package
    COMMAND mkdir ${PROJECT_BINARY_DIR}/package
    COMMAND cp -rf ${CMAKE_CURRENT_SOURCE_DIR}/root ${PROJECT_BINARY_DIR}/package
    COMMAND wgtpkg-pack -f -o ${PROJECT_BINARY_DIR}/package/${PROJECT_NAME}.wgt ${PROJECT_BINARY_DIR}/package/root)
